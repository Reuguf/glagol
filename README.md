ДОКУМЕНТАЦИЯ ПО ЯЗЫКУ ГЛАГОЛ 0.3

Глагол — русский язык программирования с файлами .гл. Он запускается командой glagol. Основная идея языка: писать программы русскими командами, но при этом иметь доступ к возможностям Python внутри.

Файл программы имеет расширение .гл. Пример имени файла: main.гл. Запуск файла: glagol main.гл. Проверка версии: glagol версия. Проверка установленного примера: glagol "%LOCALAPPDATA%\Glagol\examples\hello.гл". Проверка кода без обычного запуска: glagol проверить файл.гл. Просмотр перевода в Python: glagol файл.гл --python. Безопасный режим: glagol файл.гл --безопасно.

ОСНОВНЫЕ ПРАВИЛА СИНТАКСИСА

Обычная команда в Глаголе заканчивается точкой. Пример: выведи текст "привет". Блоки после условий, циклов, действий и событий открываются двоеточием. Внутри блока используются отступы, как в Python.

Пример:

если возраст больше или равно 18:
    выведи текст "Доступ открыт".
иначе:
    выведи текст "Доступ закрыт".

Базовое окно:

ГЛАГОЛ:

создай окно "Моё приложение" размером 500 на 300.
установи тему "тёмная".

добавь заголовок "Привет из Глагола".

когда нажата кнопка:
    если значение имя равно "":
        измени ответ на "Ты не ввёл имя".
    иначе:
        измени ответ на "Привет, " + значение имя.

запусти окно.

PYTHON:

import tkinter as tk

root = tk.Tk()
root.title("Моё приложение")
root.geometry("500x300")
root.configure(bg="#1e1e1e")

title = tk.Label(root, text="Привет из Глагола", bg="#1e1e1e", fg="white", font=("Arial", 18))
title.pack(pady=10)

имя = tk.Entry(root)
имя.pack(pady=5)

ответ = tk.Label(root, text="", bg="#1e1e1e", fg="white")
ответ.pack(pady=10)

def on_click():
    value = имя.get()
    if value == "":
        ответ.config(text="Ты не ввёл имя")
    else:
        ответ.config(text="Привет, " + value)

кнопка = tk.Button(root, text="Поздороваться", command=on_click)
кнопка.pack(pady=5)

root.mainloop()

СЕРЬЁЗНЫЙ ПРИМЕР 1: КОНСОЛЬНЫЙ МЕНЕДЖЕР ЗАДАЧ С СОХРАНЕНИЕМ В JSON

Этот пример показывает переменные, списки, словари, циклы, ввод, условия, файлы и JSON.

ГЛАГОЛ:

выведи текст "Менеджер задач на Глаголе".

если файл существует "tasks.json":
    пусть задачи будут прочитай json "tasks.json".
иначе:
    пусть задачи будут [].

пусть работает будет да.

пока работает равно да:
    выведи текст "".
    выведи текст "1 - Показать задачи".
    выведи текст "2 - Добавить задачу".
    выведи текст "3 - Сохранить задачи".
    выведи текст "0 - Выход".

    спроси текст "Выбор: " и запомни как выбор.

    если выбор равно "1":
        если len(задачи) равно 0:
            выведи текст "Задач нет".
        иначе:
            пусть номер будет 1.
            для каждого задача из задачи:
                выведи str(номер) + ". " + задача["текст"] + " | статус: " + задача["статус"].
                увеличь номер на 1.

    иначе если выбор равно "2":
        спроси текст "Текст задачи: " и запомни как текст.
        пусть задача будет {"текст": текст, "статус": "новая"}.
        задачи.append(задача).
        выведи текст "Задача добавлена".

    иначе если выбор равно "3":
        запиши json задачи в файл "tasks.json".
        выведи текст "Задачи сохранены".

    иначе если выбор равно "0":
        запиши json задачи в файл "tasks.json".
        измени работает на нет.
        выведи текст "Выход. Данные сохранены".

    иначе:
        выведи текст "Неизвестная команда".

PYTHON:

import json
from pathlib import Path

print("Менеджер задач на Python")

if Path("tasks.json").exists():
    with open("tasks.json", "r", encoding="utf-8") as file:
        задачи = json.load(file)
else:
    задачи = []

работает = True

while работает:
    print("")
    print("1 - Показать задачи")
    print("2 - Добавить задачу")
    print("3 - Сохранить задачи")
    print("0 - Выход")

    выбор = input("Выбор: ")

    if выбор == "1":
        if len(задачи) == 0:
            print("Задач нет")
        else:
            номер = 1
            for задача in задачи:
                print(str(номер) + ". " + задача["текст"] + " | статус: " + задача["статус"])
                номер += 1

    elif выбор == "2":
        текст = input("Текст задачи: ")
        задача = {"текст": текст, "статус": "новая"}
        задачи.append(задача)
        print("Задача добавлена")

    elif выбор == "3":
        with open("tasks.json", "w", encoding="utf-8") as file:
            json.dump(задачи, file, ensure_ascii=False, indent=4)
        print("Задачи сохранены")

    elif выбор == "0":

PYTHON:

import tkinter as tk
from pathlib import Path

root = tk.Tk()
root.title("Заметки на Python")
root.geometry("700x600")
root.configure(bg="#2d1b4e")

title = tk.Label(root, text="Заметки", bg="#2d1b4e", fg="white", font=("Arial", 22))
title.pack(pady=10)

description = tk.Label(root, text="Простое приложение с сохранением текста в файл", bg="#2d1b4e", fg="white")
description.pack(pady=5)

название = tk.Entry(root, width=60)
название.insert(0, "")
название.pack(pady=5)

заметка = tk.Entry(root, width=60)
заметка.insert(0, "")
заметка.pack(pady=5)

статус = tk.Label(root, text="", bg="#2d1b4e", fg="white")
статус.pack(pady=10)

лог = tk.Text(root, width=70, height=15)
лог.pack(pady=10)

def save_note():
    if название.get() == "":
        статус.config(text="Ошибка: введи название заметки")
    else:
        if заметка.get() == "":
            статус.config(text="Ошибка: заметка пустая")
        else:
            Path("note.txt").write_text(заметка.get(), encoding="utf-8")
            статус.config(text="Заметка сохранена")
            лог.delete("1.0", tk.END)
            лог.insert(tk.END, "Файл note.txt записан")

def load_note():
    if Path("note.txt").exists():
        text = Path("note.txt").read_text(encoding="utf-8")
        статус.config(text="Заметка загружена")
        лог.delete("1.0", tk.END)
        лог.insert(tk.END, text)
    else:
        статус.config(text="Файл note.txt не найден")

def clear_status():
    статус.config(text="")
    лог.delete("1.0", tk.END)

tk.Button(root, text="Сохранить", command=save_note).pack(pady=5)
tk.Button(root, text="Загрузить", command=load_note).pack(pady=5)
tk.Button(root, text="Очистить статус", command=clear_status).pack(pady=5)

root.mainloop()

import random as рандом
import string as строки

символы = строки.ascii_letters + строки.digits + "!@#$%^&*"

for _ in range(16):

    пароль = пароль + символ

print("Пароль: " + пароль)
